# 垃圾回收模块 (erwind_gc)

## 功能概述

垃圾回收模块负责磁盘队列的清理、压缩和存储配额管理。它是从原 `erwind_diskqueue` 中拆分出来的独立服务，实现存储资源的自动化管理。

## 核心职责

1. **过期消息清理**：删除超过 TTL 的消息
2. **文件合并**：合并小文件减少碎片化
3. **存储配额**：限制总存储使用量
4. **压缩归档**：历史数据压缩存储

## 为什么需要独立 GC 模块

### 原设计问题

```erlang
%% 原 erwind_diskqueue 中的清理逻辑
-record(state, {
    %% ... 其他字段
    need_cleanup = false :: boolean()
}).

%% 问题：
%% 1. 清理逻辑和读写逻辑耦合
%% 2. 无法独立控制清理策略
%% 3. 大文件清理时阻塞读写
```

### 新设计优势

```erlang
%% 独立 GC 进程
- 读写和清理解耦
- 可配置多种 GC 策略
- 支持后台异步清理
- 细粒度配额控制
```

## 数据结构

```erlang
-module(erwind_gc).
-behaviour(gen_server).

%% GC 策略
-record(gc_policy, {
    %% TTL 策略
    ttl_enabled = true :: boolean(),
    msg_ttl = 86400000 :: integer(),  %% 消息存活时间（默认24小时）
    
    %% 存储配额
    quota_enabled = false :: boolean(),
    max_bytes = 10737418240 :: integer(),  %% 10GB
    max_files = 1000 :: integer(),
    
    %% 合并策略
    compaction_enabled = true :: boolean(),
    min_file_size = 10485760 :: integer(),  %% 小于10MB的文件才合并
    max_file_age = 3600000 :: integer(),     %% 文件存活超过1小时才合并
    compaction_ratio = 0.5 :: float(),       %% 有效数据比例低于50%时合并
    
    %% 归档策略
    archive_enabled = false :: boolean(),
    archive_age = 604800000 :: integer(),    %% 7天后归档
    archive_path :: string()
}).

%% 队列统计
-record(queue_stats, {
    topic :: binary(),
    channel :: binary() | undefined,
    total_bytes :: integer(),
    total_files :: integer(),
    avg_file_size :: float(),
    oldest_msg_time :: integer(),
    newest_msg_time :: integer()
}).

%% 状态
-record(state, {
    policies = #{} :: #{binary() => #gc_policy{}},  %% Topic/Channel -> Policy
    gc_timer :: reference(),
    stats :: #queue_stats{},
    current_gc_job :: {pid(), reference()} | undefined
}).
```

## 核心逻辑

### 1. TTL 清理

```erlang
%% 按消息时间戳清理过期消息
handle_info({gc_ttl, Topic, Channel}, State) ->
    Now = erlang:system_time(millisecond),
    Policy = get_policy(State, Topic, Channel),
    
    DataPath = get_data_path(Topic, Channel),
    MetaFile = filename:join(DataPath, "meta.dat"),
    
    case load_meta(MetaFile) of
        {ok, Meta} ->
            CutoffTime = Now - Policy#gc_policy.msg_ttl,
            
            %% 遍历数据文件
            {ok, Files} = file:list_dir(DataPath),
            DataFiles = lists:filter(fun(F) -> 
                string:suffix(F, ".dat") andalso F =/= "meta.dat"
            end, Files),
            
            lists:foreach(fun(File) ->
                FilePath = filename:join(DataPath, File),
                {ok, Fd} = file:open(FilePath, [read, binary]),
                
                %% 扫描文件，找出过期消息
                ExpiredPositions = scan_expired_msgs(Fd, CutoffTime, []),
                file:close(Fd),
                
                %% 重写文件（跳过过期消息）
                case length(ExpiredPositions) of
                    0 -> ok;
                    _ -> rewrite_file(FilePath, ExpiredPositions)
                end
            end, DataFiles),
            
            {noreply, State};
        {error, _} ->
            {noreply, State}
    end.

%% 扫描过期消息
scan_expired_msgs(Fd, CutoffTime, Acc) ->
    case file:read(Fd, 12) of
        {ok, <<Len:32/big, Ts:64/big>>} ->
            {ok, _Pos} = file:position(Fd, {cur, 0}),
            case Ts < CutoffTime of
                true -> 
                    file:read(Fd, Len),  %% 跳过消息体
                    scan_expired_msgs(Fd, CutoffTime, [_Pos | Acc]);
                false ->
                    file:read(Fd, Len),
                    scan_expired_msgs(Fd, CutoffTime, Acc)
            end;
        eof ->
            lists:reverse(Acc);
        {error, _} ->
            lists:reverse(Acc)
    end.

%% 重写文件（删除过期消息）
rewrite_file(FilePath, ExpiredPositions) ->
    TmpFile = FilePath ++ ".tmp",
    {ok, InFd} = file:open(FilePath, [read, binary]),
    {ok, OutFd} = file:open(TmpFile, [write, binary]),
    
    copy_skipping(InFd, OutFd, ExpiredPositions, 0),
    
    file:close(InFd),
    file:close(OutFd),
    
    %% 原子替换
    file:rename(TmpFile, FilePath).

copy_skipping(InFd, OutFd, [SkipPos | Rest], CurrentPos) ->
    %% 复制 SkipPos 之前的数据
    BytesToCopy = SkipPos - CurrentPos,
    {ok, Data} = file:read(InFd, BytesToCopy),
    file:write(OutFd, Data),
    
    %% 跳过过期消息（读取但不写入）
    {ok, <<Len:32/big, _Ts:64/big>>} = file:read(InFd, 12),
    file:read(InFd, Len),
    
    copy_skipping(InFd, OutFd, Rest, SkipPos + 12 + Len);
copy_skipping(InFd, OutFd, [], _CurrentPos) ->
    %% 复制剩余数据
    case file:read(InFd, 65536) of
        {ok, Data} ->
            file:write(OutFd, Data),
            copy_skipping(InFd, OutFd, [], 0);
        eof ->
            ok
    end.
```

### 2. 文件合并

```erlang
%% 合并小文件
handle_info({gc_compact, Topic, Channel}, State) ->
    Policy = get_policy(State, Topic, Channel),
    DataPath = get_data_path(Topic, Channel),
    
    {ok, Files} = file:list_dir(DataPath),
    DataFiles = lists:filter(fun(F) -> 
        string:suffix(F, ".dat") andalso F =/= "meta.dat"
    end, Files),
    
    %% 获取文件信息
    FileInfos = lists:map(fun(File) ->
        Path = filename:join(DataPath, File),
        {ok, Info} = file:read_file_info(Path),
        {File, Info#file_info.size, Info#file_info.mtime}
    end, DataFiles),
    
    %% 找出需要合并的小文件
    SmallFiles = lists:filter(fun({_F, Size, _Mtime}) ->
        Size < Policy#gc_policy.min_file_size
    end, FileInfos),
    
    %% 按大小排序
    SortedFiles = lists:sort(fun({_, S1, _}, {_, S2, _}) ->
        S1 < S2
    end, SmallFiles),
    
    %% 合并文件（每批合并最多10个）
    batch_merge(SortedFiles, Policy, DataPath),
    
    {noreply, State}.

%% 批量合并
batch_merge(Files, Policy, DataPath) when length(Files) >= 2 ->
    {Batch, Rest} = lists:split(min(10, length(Files)), Files),
    
    TotalSize = lists:sum([S || {_, S, _} <- Batch]),
    
    %% 检查合并后是否合理
    case TotalSize < Policy#gc_policy.min_file_size * 5 of
        true ->
            %% 执行合并
            merge_files(Batch, DataPath),
            batch_merge(Rest, Policy, DataPath);
        false ->
            batch_merge(Rest, Policy, DataPath)
    end;
batch_merge(_, _, _) ->
    ok.

%% 合并多个文件到一个新文件
merge_files(Files, DataPath) ->
    [FirstFile | Rest] = [F || {F, _, _} <- Files],
    
    %% 使用第一个文件名，其他文件内容追加
    TargetFile = filename:join(DataPath, FirstFile),
    {ok, OutFd} = file:open(TargetFile, [append, binary]),
    
    lists:foreach(fun({File, _, _}) ->
        FilePath = filename:join(DataPath, File),
        {ok, InFd} = file:open(FilePath, [read, binary]),
        
        %% 复制内容
        copy_all(InFd, OutFd),
        
        file:close(InFd),
        
        %% 删除原文件（除了目标文件）
        case File =/= FirstFile of
            true -> file:delete(FilePath);
            false -> ok
        end
    end, Rest),
    
    file:close(OutFd).

copy_all(InFd, OutFd) ->
    case file:read(InFd, 65536) of
        {ok, Data} ->
            file:write(OutFd, Data),
            copy_all(InFd, OutFd);
        eof ->
            ok
    end.
```

### 3. 存储配额管理

```erlang
%% 检查并强制执行存储配额
handle_info(check_quota, State) ->
    StatsList = collect_all_stats(),
    
    lists:foreach(fun(Stats) ->
        Topic = Stats#queue_stats.topic,
        Channel = Stats#queue_stats.channel,
        Policy = get_policy(State, Topic, Channel),
        
        case Policy#gc_policy.quota_enabled of
            true ->
                enforce_quota(Stats, Policy);
            false ->
                ok
        end
    end, StatsList),
    
    %% 每小时检查一次
    erlang:send_after(3600000, self(), check_quota),
    
    {noreply, State}.

%% 强制执行配额
enforce_quota(Stats, Policy) ->
    %% 检查字节配额
    case Stats#queue_stats.total_bytes > Policy#gc_policy.max_bytes of
        true ->
            BytesToFree = Stats#queue_stats.total_bytes - 
                         Policy#gc_policy.max_bytes,
            free_space(Stats, BytesToFree);
        false ->
            ok
    end,
    
    %% 检查文件数配额
    case Stats#queue_stats.total_files > Policy#gc_policy.max_files of
        true ->
            FilesToDelete = Stats#queue_stats.total_files - 
                           Policy#gc_policy.max_files,
            delete_oldest_files(Stats, FilesToDelete);
        false ->
            ok
    end.

%% 释放空间（从最老的文件开始删）
free_space(Stats, BytesToFree) ->
    DataPath = get_data_path(Stats#queue_stats.topic, 
                            Stats#queue_stats.channel),
    
    %% 获取文件按时间排序
    {ok, Files} = file:list_dir(DataPath),
    FileInfos = lists:map(fun(File) ->
        Path = filename:join(DataPath, File),
        {ok, Info} = file:read_file_info(Path),
        {File, Info#file_info.size, Info#file_info.mtime}
    end, Files),
    
    SortedFiles = lists:sort(fun({_, _, T1}, {_, _, T2}) ->
        T1 < T2
    end, FileInfos),
    
    %% 删除最老的文件直到释放足够空间
    free_space_loop(SortedFiles, BytesToFree, DataPath).

free_space_loop([{File, Size, _} | Rest], Remaining, DataPath) when Remaining > 0 ->
    FilePath = filename:join(DataPath, File),
    file:delete(FilePath),
    free_space_loop(Rest, Remaining - Size, DataPath);
free_space_loop(_, _, _) ->
    ok.
```

### 4. 后台 GC 任务

```erlang
%% 启动后台 GC（避免阻塞主进程）
handle_call({trigger_gc, Topic, Channel}, _From, State) ->
    JobId = make_ref(),
    
    %% 启动子进程执行 GC
    Pid = spawn_link(fun() ->
        perform_gc(Topic, Channel, State#state.policies)
    end),
    
    NewState = State#state{
        current_gc_job = {Pid, JobId}
    },
    
    {reply, {ok, JobId}, NewState}.

%% 后台 GC 执行
perform_gc(Topic, Channel, Policies) ->
    Policy = maps:get({Topic, Channel}, Policies, #gc_policy{}),
    
    %% 1. TTL 清理
    gc_ttl(Topic, Channel, Policy),
    
    %% 2. 文件合并
    gc_compact(Topic, Channel, Policy),
    
    %% 3. 归档处理
    gc_archive(Topic, Channel, Policy),
    
    %% 通知完成
    erwind_gc ! {gc_complete, Topic, Channel}.

handle_info({gc_complete, Topic, Channel}, State) ->
    error_logger:info_msg("GC completed for ~p/~p", [Topic, Channel]),
    {noreply, State#state{current_gc_job = undefined}}.
```

## 最佳实践

1. **TTL 设置**：根据业务场景，通常 24-72 小时
2. **合并策略**：小文件(<10MB)且旧文件(>1小时)才合并
3. **配额预警**：使用率达 80% 时触发告警
4. **后台执行**：大文件清理使用子进程避免阻塞
